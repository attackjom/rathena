ra_fild12,295,208,0	script	toairplane#rachel	45,1,1,{
	end;
OnTouch:
	mes "ในการใช้เรือเหาะ ท่านจะต้องจ่าย 1,200 zeny หรือ ใช้ตั๋วเรือเหาะฟรี";
	mes "ท่านต้องการใช้บริการหรือไม่?";
	next;
	if (select("ใช่:ไม่") == 1) {
		if (countitem(7311) > 0) {
			delitem 7311,1; //Free_Flying_Ship_Ticket
			warp "airplane_01",245,60;
			end;
		}
		if (Zeny >= 1200) {
			set Zeny, Zeny - 1200;
			warp "airplane_01",245,60;
			end;
		}
		mes "ขออภัย แต่ท่านไม่มีเงินเพียงพอ";
		mes "โปรดจำไว้ว่า ท่านจะต้องจ่าย 1,200 zeny เพื่อใช้บริการ";
		close;
	}
	mes "ขอบคุณ โปรดกลับมาอีกครั้ง";
	close;
}
