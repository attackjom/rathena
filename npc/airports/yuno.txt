y_airport,143,43,5	script	Airport Staff#y_air1a::y_airport1	90,{
	mes "[Airport Staff]";
	mes "ยินดีต้อนรับสู่สนามบิน Juno เราให้บริการเที่ยวบินภายในประเทศไปยัง Einbroch, Lighthalzen และ Hugel";
	mes "และเที่ยวบินระหว่างประเทศไปยัง Izlude และ Rachel";
	mes "ข้าจะให้บริการได้อย่างไร?";
	next;
	if (select("ขึ้นเรือเหาะ:ยกเลิก") == 1) {
		mes "[Airport Staff]";
		mes "ค่าธรรมเนียมการขึ้นเรือเหาะ";
		mes "สำหรับทุกเที่ยวบินคือ 1,200 zeny";
		mes "หากท่านใช้ตั๋วฟรีสำหรับเรือเหาะ";
		mes "ค่าธรรมเนียมการขึ้นเรือเหาะจะถูกยกเลิก ดังนั้นท่านต้องการออกเดินทาง?";
		next;
		if (select("ใช่:ไม่") == 1) {
			if (countitem(7311) > 0) {
				delitem 7311,1; //Free_Flying_Ship_Ticket
				warp "y_airport",148,51;
				end;
			}
			if (Zeny >= 1200) {
				set Zeny, Zeny - 1200;
				warp "y_airport",148,51;
				end;
			}
			mes "[Airport Staff]";
			mes "ขออภัย แต่ท่าน";
			mes "ไม่มีเงิน 1,200 zeny เพื่อ";
			mes "ชำระค่าธรรมเนียมการขึ้นเครื่อง";
			close;
		}
	}
	mes "[Airport Staff]";
	mes "ขอบคุณ และ";
	mes "ขอให้มีความสุข";
	close;
}
