lhz_airport,143,43,5	script	Airport Staff#lhz_air1a::lhz_airport1	90,{
	mes "[Airport Staff]";
	mes "ยินดีต้อนรับสู่";
	mes "สนามบิน Lighthalzen";
	mes "เราให้บริการเที่ยวบินแบบไม่หยุดพัก";
	mes "เพื่อไปยัง Einbroch, Juno และ Hugel";
	next;
	if (select("ขึ้นเรือเหาะ:ยกเลิก") == 1) {
		mes "[Airport Staff]";
		mes "ค่าธรรมเนียมการขึ้นเรือเหาะ";
		mes "คือ 1,200 zeny แต่ท่าน";
		mes "สามารถยกเว้นค่าธรรมเนียมได้";
		mes "ถ้าท่านใช้ตั๋วฟรีสำหรับเรือเหาะ";
		next;
		if (select("ใช่:ไม่") == 1) {
			if (countitem(7311) > 0) {
				delitem 7311,1; //Free_Flying_Ship_Ticket
				warp "lhz_airport",148,51;
				end;
			}
			if (Zeny >= 1200) {
				set Zeny, Zeny - 1200;
				warp "lhz_airport",148,51;
				end;
			}
			mes "[Airship Staff]";
			mes "ขออภัย เเต่ท่าน";
			mes "ไม่มีเงิน 1,200 zeny เพื่อ";
			mes "ชำระค่าธรรมเนียมการขึ้นเครื่อง";
			close;
		}
	}
	mes "[Airport Staff]";
	mes "ขอบคุณ และ";
	mes "โปรดกลับมาอีกครั้ง";
	mes "ขอให้เป็นวันที่ดี~";
	close;
}
