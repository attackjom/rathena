airport,143,43,5	script	Airport Staff#airport1a::airport1	90,{
	mes "[Airport Staff]";
	mes "ยินดีต้อนรับสู่";
	mes "สนามบิน Einbroch";
	mes "เราให้บริการเที่ยวบินแบบ";
	mes "ไม่หยุดพัก เพื่อไปยังเมืองต่าง ๆ";
	mes "เช่น Juno, Lighthalzen เเละ Hugel";
	next;
	if (select("ขึ้นเรือเหาะ:ยกเลิก") == 1) {
		mes "[Airport Staff]";
		mes "ค่าธรรมเนียมการขึ้น";
		mes "เรือเหาะ 1,200 zeny แต่";
		mes "ถ้าท่านมีตั๋วฟรีสำหรับเรือเหาะ";
		mes "ค่าธรรมเนียมจะถูกยกเลิก";
		mes "ท่านจะขึ้นเรือเหาะไหม?";
		next;
		if (select("ใช่:ไม่") == 1) {
			if (countitem(7311) > 0) {
				delitem 7311,1; //Free_Flying_Ship_Ticket
				warp "airport",148,51;
				end;
			}
			if (Zeny >= 1200) {
				set Zeny, Zeny - 1200;
				warp "airport",148,51;
				end;
			}
			mes "[Airport Staff]";
			mes "ขออภัย แต่";
			mes "ท่านไม่มีตั๋วฟรีสำหรับ";
			mes "ขึ้นเรือเหาะ และท่านไม่มี";
			mes "เงินมากพอ สำหรับการขึ้นบิน";
			mes "จำไว้ว่า ค่าขึ้นเครื่อง";
			mes "คือ 1,200 zeny";
			close;
		}
	}
	mes "[Airport Staff]";
	mes "ขอบคุณ และ";
	mes "ขอให้เป็นวันที่ดี";
	close;
}
