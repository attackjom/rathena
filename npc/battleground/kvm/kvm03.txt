-	script	KvM03_BG	-1,{
	end;

OnInit:
	setwall "bat_c03",54,122,6,7,0,"batc03wall_a";
	setwall "bat_c03",55,122,5,7,0,"batc03wall_b";
	setwall "bat_c03",140,56,6,7,0,"batc03wall_c";
	setwall "bat_c03",140,57,5,7,0,"batc03wall_d";
	disablenpc "KVM Officer#KVM03A";
	disablenpc "KVM Officer#KVM03B";
	end;

OnGuillaumeQuit:
	if (getbattleflag("feature.bgqueue"))
		bg_desert;
//	else
//		set BG_Delay_Tick, gettimetick(2) + 1200;
OnGuillaumeDie:
	if( $@KvM03BG == 2 ) {
		set .Guillaume_Count, .Guillaume_Count - 1;
		bg_updatescore "bat_c03",.Guillaume_Count,.Croix_Count;
		if( .Guillaume_Count < 1 )
			donpcevent "KvM03_BG::OnCroixWin";
		else {
			mapannounce "bat_c03", "จำนวน Guillaumes คือ "+.Guillaume_Count+".",bc_map,"0x00ff00";
			mapannounce "bat_c03", "จำนวน Croixes คือ "+.Croix_Count+".",bc_map,"0x00ff00";
		}
	}
	end;

OnCroixQuit:
	if (getbattleflag("feature.bgqueue"))
		bg_desert;
//	else
//	set BG_Delay_Tick, gettimetick(2) + 1200;
OnCroixDie:
	if( $@KvM03BG == 2 ) {
		set .Croix_Count, .Croix_Count - 1;
		bg_updatescore "bat_c03",.Guillaume_Count,.Croix_Count;
		if( .Croix_Count < 1 )
			donpcevent "KvM03_BG::OnGuillaumeWin";
		else {
			mapannounce "bat_c03", "จำนวน Guillaumes คือ "+.Guillaume_Count+".",bc_map,"0x00ff00";
			mapannounce "bat_c03", "จำนวน Croixes คือ "+.Croix_Count+".",bc_map,"0x00ff00";
		}
	}
	end;

OnGuillaumeActive:
	warp "bat_c03",62,119;
	end;

OnCroixActive:
	warp "bat_c03",137,64;
	end;

OnStart:
	disablenpc "KVM Officer#KVM03A";
	disablenpc "KVM Officer#KVM03B";
	set $@KvM03BG_Victory, 0;
	// Warp Teams
	bg_warp $@KvM03BG_id1,"bat_c03",53,128;
	bg_warp $@KvM03BG_id2,"bat_c03",146,55;
	initnpctimer;
	end;

OnTimer1000:
	mapannounce "bat_c03", "ภายใน 1 นาที KVM จะเริ่มต้น",bc_map,"0x00ff00";
	end;

OnTimer3000:
	mapannounce "bat_c03", "เวลาสูงสุดสำหรับการต่อสู้ KVM คือ 5 นาที",bc_map,"0x00ff00";
	end;

OnTimer6000:
	mapannounce "bat_c03", "โปรดเตรียมความพร้อม สำหรับการต่อสู้ KVM",bc_map,"0x00ff00";
	mapannounce "bat_c03", "ท่านสามารถเพิ่มความสามารถพิเศษให้คนของท่านได้",bc_map,"0x00ff00";
	donpcevent "#A_camp_start03::OnEnable";
	donpcevent "#B_camp_start03::OnEnable";
	end;

OnTimer13000:
	donpcevent "#A_camp_start03::OnDisable";
	donpcevent "#B_camp_start03::OnDisable";
	end;

OnTimer30000:
	mapannounce "bat_c03", "อีก 30 วินาที การต่อสู้ KVM จะเริ่มต้น",bc_map,"0x00ff00";
	end;

OnTimer45000:
	mapannounce "bat_c03", "อีก 15 วินาที การต่อสู้ KVM จะเริ่มต้น",bc_map,"0x00ff00";
	donpcevent "::OnKvM03One";
	end;

OnTimer50000:
	mapannounce "bat_c03", "อีก 10 วินาที การต่อสู้ KVM จะเริ่มต้น",bc_map,"0x00ff00";
	donpcevent "::OnKvM03Two";
	end;

OnTimer55000:
	mapannounce "bat_c03", "อีก 5 วินาที การต่อสู้ KVM จะเริ่มต้น",bc_map,"0x00ff00";
	donpcevent "::OnKvM03One";
	end;

OnTimer59000:
	mapannounce "bat_c03", "KVM ได้เริ่มขึ้นแล้ว",bc_map,"0x00ff00";
	donpcevent "::OnKvM03Two";
	end;

OnTimer61000:
	// Team Members
	set .Guillaume_Count, bg_get_data($@KvM03BG_id1, 0);
	set .Croix_Count, bg_get_data($@KvM03BG_id2, 0);

	if (!getbattleflag("feature.bgqueue")) {
		if (.Guillaume_Count < 5 || .Croix_Count < 5) {
			set $@KvM03BG_Victory, 3;
			set $@KvM03BG, 3;
			mapannounce "bat_c03","มีผู้เล่นไม่เพียงพอ ที่จะเริ่มการต่อสู้",1,0xC0C0C0;
			stopnpctimer;
			donpcevent "KvM03_BG::OnStop";
			end;
		}
	}
	bg_updatescore "bat_c03",.Guillaume_Count,.Croix_Count;
	set $@KvM03BG, 2; // Playing
	bg_warp $@KvM03BG_id1,"bat_c03",62,119;
	bg_warp $@KvM03BG_id2,"bat_c03",137,64;
	end;

OnTimer300000:
	mapannounce "bat_c03", "อีก 1 นาที การต่อสู้ KVM จะสิ้นสุด",bc_map,"0x00ff00";
	end;

OnTimer330000:
	mapannounce "bat_c03", "อีก 30 นาที การต่อสู้ KVM จะสิ้นสุด",bc_map,"0x00ff00";
	end;

OnTimer345000:
	mapannounce "bat_c03", "อีก 15 นาที การต่อสู้ KVM จะสิ้นสุด",bc_map,"0x00ff00";
	end;

OnTimer350000:
	mapannounce "bat_c03", "อีก 10 นาที การต่อสู้ KVM จะสิ้นสุด",bc_map,"0x00ff00";
	end;

OnTimer355000:
	mapannounce "bat_c03", "อีก 5 นาที การต่อสู้ KVM จะสิ้นสุด",bc_map,"0x00ff00";
	end;

OnTimer360000:
	mapannounce "bat_c03", "การต่อสู้ KVM จบลงแล้ว",bc_map,"0x00ff00";
	if( .Croix_Count > .Guillaume_Count )
		donpcevent "KvM03_BG::OnCroixWin";
	else if( .Croix_Count < .Guillaume_Count )
		donpcevent "KvM03_BG::OnGuillaumeWin";
	else
	{ // Draw Game
		set $@KvM03BG, 3;
		set $@KvM03BG_Victory, 3;
		mapannounce "bat_c03", "จำนวน Guillaumes คือ "+.Guillaume_Count+".",bc_map,"0x00ff00";
		mapannounce "bat_c03", "จำนวน Croixes คือ "+.Croix_Count+".",bc_map,"0x00ff00";
		mapannounce "bat_c03", "การต่อสู้ครั้งนี้จบลงด้วยผลเสมอ",bc_map,"0x00ff00";
		donpcevent "KvM03_BG::OnStop";
	}
	end;

OnGuillaumeWin:
	set $@KvM03BG, 3;
	set $@KvM03BG_Victory, 1;
	mapannounce "bat_c03", "Guillaume ชนะ!",bc_map,"0x00ff00";
	mapannounce "bat_c03", "ขอแสดงความยินดีกับสมาชิก Guillaume",bc_map,"0x00ff00";
	mapannounce "bat_c03", "ทุกคนจะถูกย้ายไปยังจุดเริ่มต้น",bc_map,"0x00ff00";
	donpcevent "KvM03_BG::OnStop";
	end;

OnCroixWin:
	set $@KvM03BG, 3;
	set $@KvM03BG_Victory, 2;
	mapannounce "bat_c03", "Croix ชนะ!",bc_map,"0x00ff00";
	mapannounce "bat_c03", "ขอแสดงความยินดีกับสมาชิก Croix",bc_map,"0x00ff00";
	mapannounce "bat_c03", "ทุกคนจะถูกย้ายไปยังจุดเริ่มต้น",bc_map,"0x00ff00";
	donpcevent "KvM03_BG::OnStop";
	end;

OnStop:
	stopnpctimer;
	enablenpc "KVM Officer#KVM03A";
	enablenpc "KVM Officer#KVM03B";
	// Warp Teams
	bg_warp $@KvM03BG_id1,"bat_c03",53,128;
	bg_warp $@KvM03BG_id2,"bat_c03",146,55;
	donpcevent "KvM03_BG_Out::OnBegin";
	if (getbattleflag("feature.bgqueue"))
		bg_reserve "bat_c03", true;
	end;
}

	set getvariableofnpc(.Croix_Count,"KvM03_BG"), 0;
	set getvariableofnpc(.Guillaume_Count,"KvM03_BG"), 0;
	set $@KvM03BG_Victory, 0;
	if( $@KvM03BG_id1 ) { bg_destroy $@KvM03BG_id1; set $@KvM03BG_id1, 0; }
	if( $@KvM03BG_id2 ) { bg_destroy $@KvM03BG_id2; set $@KvM03BG_id2, 0; }
	if (getbattleflag("feature.bgqueue"))
		bg_unbook "bat_c03";
	disablenpc "KVM Officer#KVM03A";
	disablenpc "KVM Officer#KVM03B";
	mapwarp "bat_c03","bat_room",154,150;
	maprespawnguildid "bat_c03",0,3; // Just in case someone else
	set $@KvM03BG, 0;

OnGuillaumeJoin:
OnCroixJoin:
	if( $@KvM03BG )
		end;
	set .@Guillaume, getwaitingroomstate(0,"KvM03R_Guillaume");
	set .@Croix, getwaitingroomstate(0,"KvM03R_Croix");

	if( .@Guillaume < 5 || .@Croix < 5 )
		end;

	set $@KvM03BG, 1; // Starting
	donpcevent "KvM03R_Croix::OnEnterBG";
	donpcevent "KvM03R_Guillaume::OnEnterBG";
	donpcevent "KvM03_BG::OnStart";
	end;
}

	end;
